-- 1. Таблица сотрудников
CREATE TABLE employees (
    employee_id SERIAL PRIMARY KEY,
    first_name VARCHAR(50) NOT NULL,
    last_name VARCHAR(50) NOT NULL,
    position VARCHAR(100),
    email VARCHAR(100) UNIQUE NOT NULL
);

-- 2. Таблица проектов
CREATE TABLE projects (
    project_id SERIAL PRIMARY KEY,
    project_name VARCHAR(150) NOT NULL,
    description TEXT,
    start_date DATE NOT NULL
);

-- 3. Таблица задач
CREATE TABLE tasks (
    task_id SERIAL PRIMARY KEY,
    project_id INT NOT NULL,
    employee_id INT,
    task_name VARCHAR(200) NOT NULL,
    status VARCHAR(50) DEFAULT 'Новая',
    CONSTRAINT fk_task_project FOREIGN KEY (project_id) REFERENCES projects(project_id) ON DELETE CASCADE,
    CONSTRAINT fk_task_employee FOREIGN KEY (employee_id) REFERENCES employees(employee_id) ON DELETE SET NULL
);

-- 4. Таблица отметок времени
CREATE TABLE time_logs (
    log_id SERIAL PRIMARY KEY,
    employee_id INT NOT NULL,
    task_id INT NOT NULL,
    start_time TIMESTAMP NOT NULL,
    end_time TIMESTAMP,
    CONSTRAINT fk_log_employee FOREIGN KEY (employee_id) REFERENCES employees(employee_id) ON DELETE CASCADE,
    CONSTRAINT fk_log_task FOREIGN KEY (task_id) REFERENCES tasks(task_id) ON DELETE CASCADE
);

-- 5. Таблица отчетов
CREATE TABLE reports (
    report_id SERIAL PRIMARY KEY,
    project_id INT NOT NULL,
    created_by INT NOT NULL,
    report_date DATE NOT NULL DEFAULT CURRENT_DATE,
    content TEXT NOT NULL,
    CONSTRAINT fk_report_project FOREIGN KEY (project_id) REFERENCES projects(project_id) ON DELETE CASCADE,
    CONSTRAINT fk_report_employee FOREIGN KEY (created_by) REFERENCES employees(employee_id) ON DELETE CASCADE
);
